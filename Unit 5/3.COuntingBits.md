```
class Solution {
    public int[] countBits(int n) {
       int[] ans= new int[n+1];
       ans[0]=0;
       if(n>=1)
       {
        ans[1]=1;
       } 

       for(int x=2;x<=n;x++)
       {
        if(x%2==0)
        {
            ans[x]=ans[x/2];
        }
        else
        {
            ans[x]=ans[x/2]+1;
        }
       }
       return ans;
    }
}
```
**SPACE COMPLEXITY:O(N)**
**TIME COMPLEXITY: O(N)**
